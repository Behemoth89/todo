```typescript
import { prisma } from './prisma';

/**
 * Calculate whether a todo is ready to execute.
 * 
 * Ready to execute when:
 * - No shopping items exist OR all shopping items are bought
 * AND
 * - No parent todos exist OR all parent todos are completed
 */
export async function calculateReadyToExecute(todoId: string): Promise<boolean> {
  // Check shopping items
  const shoppingItems = await prisma.shoppingItem.findMany({
    where: { todoId, deletedAt: null },
    select: { boughtAt: true },
  });
  
  const allShoppingBought = shoppingItems.length === 0 || 
    shoppingItems.every(item => item.boughtAt !== null);
  
  // Check parent todos
  const todo = await prisma.todo.findUnique({
    where: { id: todoId },
    select: { parentId: true },
  });
  
  if (!todo || !todo.parentId) {
    // No parent = no blocking
    return allShoppingBought;
  }
  
  // Check if parent is completed
  const parent = await prisma.todo.findUnique({
    where: { id: todo.parentId },
    select: { completedAt: true, deletedAt: true },
  });
  
  // If parent is deleted, it's not a blocker
  if (!parent || parent.deletedAt) {
    return allShoppingBought;
  }
  
  const parentCompleted = parent.completedAt !== null;
  
  return allShoppingBought && parentCompleted;
}

/**
 * Recalculate ready status for a todo and all its children.
 * Called after shopping item changes or parent completion changes.
 */
export async function recalculateReadyStatus(todoId: string): Promise<void> {
  // Calculate for the todo itself
  const ready = await calculateReadyToExecute(todoId);
  
  await prisma.todo.update({
    where: { id: todoId },
    data: { readyToExecute: ready },
  });
  
  // Find all children that have this todo as parent
  const children = await prisma.todo.findMany({
    where: { parentId: todoId },
    select: { id: true },
  });
  
  // Recursively update children
  for (const child of children) {
    await recalculateReadyStatus(child.id);
  }
}

/**
 * Detect circular dependency when setting parentId.
 * Returns true if setting newParentId as parent of todoId would create a cycle.
 */
export async function detectCycle(todoId: string, newParentId: string): Promise<boolean> {
  // If the potential parent is the todo itself, it's a cycle
  if (todoId === newParentId) return true;
  
  // Walk up the parent chain from newParentId
  let currentId: string | null = newParentId;
  
  while (currentId) {
    if (currentId === todoId) {
      return true; // Cycle found - todoId would become descendant of itself
    }
    
    const current = await prisma.todo.findUnique({
      where: { id: currentId },
      select: { parentId: true },
    });
    
    currentId = current?.parentId || null;
  }
  
  return false;
}
```
